By Rehan January 16, 2024 12 min read

Project Overview

In this tutorial, we'll create a feature-rich blog application using Vue 3's Composition API. We'll implement a clean, modern design with all the essential blog features.

We'll build a blog with:

  • A responsive layout with header, sidebar, and main content
  • Dynamic blog post listing with filtering
  • Interactive post details with comments
  • A rich text editor for creating posts
  • Dark/light theme toggle
  • State management using Vue 3's reactive system

3. Composable: useBlogPosts

import { ref, computed } from 'vue'
import { useStorage } from '@vueuse/core'

export function useBlogPosts() {
  // Use localStorage for persistence with @vueuse/core
  const posts = useStorage('blog-posts', [
    {
      id: 1,
      title: "Getting Started with Vue 3 Composition API",
      content: "The Composition API is a new way to organize and reuse logic in Vue 3 components...",
      excerpt: "Learn the basics of Vue 3's Composition API and how it revolutionizes component organization.",
      author: "Jane Developer",
      date: "2023-10-15",
      category: "Vue.js",
      tags: ["Vue 3", "Composition API", "Frontend"],
      likes: 42,
      comments: 12,
      featured: true
    },
    // More posts...
  ])

  const loading = ref(false)
  const error = ref(null)
  
  // Computed properties
  const categories = computed(() => {
    const allCategories = posts.value.map(post => post.category)
    return ['all', ...new Set(allCategories)]
  })
  
  const featuredPosts = computed(() => {
    return posts.value.filter(post => post.featured)
  })
  
  const totalPosts = computed(() => posts.value.length)
  
  // Methods
  const fetchPosts = async () => {
    loading.value = true
    try {
      // In a real app, you would fetch from an API
      // Simulate API call delay
      await new Promise(resolve => setTimeout(resolve, 300))
      loading.value = false
    } catch (err) {
      error.value = err.message
      loading.value = false
    }
  }
  
  const getPostById = (id) => {
    return posts.value.find(post => post.id === id)
  }
  
  const addPost = (post) => {
    const newPost = {
      ...post,
      id: posts.value.length + 1,
      date: new Date().toISOString().split('T')[0],
      likes: 0,
      comments: 0
    }
    posts.value.unshift(newPost)
    return newPost
  }
  
  const updatePost = (id, updatedData) => {
    const index = posts.value.findIndex(post => post.id === id)
    if (index !== -1) {
      posts.value[index] = { ...posts.value[index], ...updatedData }
    }
  }
  
  const deletePost = (id) => {
    const index = posts.value.findIndex(post => post.id === id)
    if (index !== -1) {
      posts.value.splice(index, 1)
    }
  }
  
  const searchPosts = (query) => {
    if (!query.trim()) return posts.value
    const lowercaseQuery = query.toLowerCase()
    return posts.value.filter(post => 
      post.title.toLowerCase().includes(lowercaseQuery) ||
      post.content.toLowerCase().includes(lowercaseQuery) ||
      post.tags.some(tag => tag.toLowerCase().includes(lowercaseQuery))
    )
  }
  
  const likePost = (id) => {
    const post = getPostById(id)
    if (post) {
      post.likes++
    }
  }
  
  return {
    posts,
    loading,
    error,
    categories,
    featuredPosts,
    totalPosts,
    fetchPosts,
    getPostById,
    addPost,
    updatePost,
    deletePost,
    searchPosts,
    likePost
  }
}

import { computed } from 'vue'

  categoryFilter: {
    type: String,
    default: 'all'
  },
  searchQuery: {
    type: String,
    default: ''
  }
})

const filteredPosts = computed(() => {
  let filtered = posts.value
  
  if (props.categoryFilter !== 'all') {
    filtered = filtered.filter(post => post.category === props.categoryFilter)
  }
  
  if (props.searchQuery) {
    const query = props.searchQuery.toLowerCase()
    filtered = filtered.filter(post => 
      post.title.toLowerCase().includes(query) ||
      post.content.toLowerCase().includes(query) ||
      post.tags.some(tag => tag.toLowerCase().includes(query))
    )
  }
  
  return filtered
})

  router.push(`/post/${id}`)

5. Composable: useTheme

import { ref, onMounted, watch } from 'vue'
import { useStorage } from '@vueuse/core'

export function useTheme() {
  const isDarkTheme = useStorage('dark-theme', false)
  
  const toggleTheme = () => {
    isDarkTheme.value = !isDarkTheme.value
    applyTheme()
  }
  
  const applyTheme = () => {
    if (isDarkTheme.value) {
      document.documentElement.classList.add('dark-theme')
    } else {
      document.documentElement.classList.remove('dark-theme')
    }
  }
  
  // Apply theme on mount and when it changes
  onMounted(applyTheme)
  watch(isDarkTheme, applyTheme)
  
  return {
    isDarkTheme,
    toggleTheme
  }
}

<template>
  <div class="editor-container">
    <div class="editor-toolbar">
      <button @click="formatText('bold')" :class="{ active: formats.bold }">
        <strong>B</strong>
      </button>

        <em>I</em>
      </button>
      <button @click="formatText('underline')" :class="{ active: formats.underline }">
        <u>U</u>
      </button>
      <button @click="insertLink">üîó</button>
      <button @click="insertImage">üñºÔ∏è</button>
      <select v-model="headingLevel" @change="formatHeading">
        <option value="">Normal</option>
        <option value="h1">Heading 1</option>
        <option value="h2">Heading 2</option>
        <option value="h3">Heading 3</option>
      </select>
    </div>
    
    <div 
      ref="editorRef"
      class="editor-content"
      contenteditable="true"
      @input="handleInput"
      @focus="handleFocus"
      @blur="handleBlur"
      @mouseup="updateFormats"
      @keyup="updateFormats"
      :placeholder="placeholder"
    ></div>
    
    <div class="editor-footer">
      <div class="char-count">
        Characters: {{ characterCount }} | Words: {{ wordCount }}
      </div>
      <div class="editor-actions">
        <button @click="clearContent" class="secondary-btn">Clear</button>
        <button @click="emitContent" class="primary-btn">Insert</button>
      </div>
    </div>
  </div>
</template>

<script setup>
import { ref, computed, onMounted } from 'vue'

const props = defineProps({
  placeholder: {
    type: String,
    default: 'Start writing your blog post here...'
  },
  initialContent: {
    type: String,
    default: ''
  }
})

const emit = defineEmits(['content-updated', 'insert'])

const editorRef = ref(null)
const headingLevel = ref('')
const formats = ref({
  bold: false,
  italic: false,
  underline: false
})

const characterCount = computed(() => {
  return getContent().length
})

const wordCount = computed(() => {
  const text = getContent().trim()
  return text.length ? text.split(/\s+/).length : 0
})

const getContent = () => {
  return editorRef.value ? editorRef.value.innerText : ''
}

const getHTMLContent = () => {
  return editorRef.value ? editorRef.value.innerHTML : ''
}

const handleInput = () => {
  emit('content-updated', {
    text: getContent(),
    html: getHTMLContent()
  })
}

const formatText = (format) => {
  document.execCommand(format, false)
  updateFormats()
}

const formatHeading = () => {
  if (headingLevel.value) {
    document.execCommand('formatBlock', false, headingLevel.value)
  } else {
    document.execCommand('formatBlock', false, '<div>')
  }
}

const insertLink = () => {
  const url = prompt('Enter URL:')
  if (url) {
    document.execCommand('createLink', false, url)
  }
}

const insertImage = () => {
  const url = prompt('Enter image URL:')
  if (url) {
    document.execCommand('insertImage', false, url)
  }
}

const updateFormats = () => {
  if (!editorRef.value) return
  
  formats.value.bold = document.queryCommandState('bold')
  formats.value.italic = document.queryCommandState('italic')
  formats.value.underline = document.queryCommandState('underline')
}

const handleFocus = () => {
  editorRef.value.classList.add('focused')
}

const handleBlur = () => {
  editorRef.value.classList.remove('focused')
}

const clearContent = () => {
  if (editorRef.value) {
    editorRef.value.innerHTML = ''
    handleInput()
  }
}

const emitContent = () => {
  emit('insert', {
    text: getContent(),
    html: getHTMLContent()
  })
}

// Set initial content if provided
onMounted(() => {
  if (props.initialContent && editorRef.value) {
    editorRef.value.innerHTML = props.initialContent
  }
})
</script>

7. Main.js Setup

import { createApp } from 'vue'
import { createRouter, createWebHistory } from 'vue-router'
import App from './App.vue'

// Import components for routes
import PostList from './components/PostList.vue'
import PostDetail from './components/PostDetail.vue'
import CreatePost from './components/CreatePost.vue'

// Create router
const router = createRouter({
  history: createWebHistory(),
  routes: [
    { path: '/', component: PostList },
    { path: '/post/:id', component: PostDetail, props: true },
    { path: '/create-post', component: CreatePost },
    { path: '/category/:category', component: PostList, props: true }
  ]
})

// Create and mount the app
const app = createApp(App)
app.use(router)
app.mount('#app')

Key Vue 3 Composition API Features Demonstrated

  1. <script setup> Syntax: Cleaner component setup without the need for explicit setup() function
  2. Reactive State: Using ref() and reactive() for reactive data
  3. Computed Properties: Using computed() for derived reactive values
  4. Composables: Reusable logic in useBlogPosts.js and useTheme.js
  5. Lifecycle Hooks: Using onMounted() for side effects
  6. Watchers: Using watch() to react to theme changes
  7. Template Refs: Using ref to access DOM elements
  8. Props and Emits: Component communication with defineProps() and defineEmits()
